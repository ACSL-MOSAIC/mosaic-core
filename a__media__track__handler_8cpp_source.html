101 static webrtc::VideoFrame CvMat2VideoFrame(const cv::Mat& frame, std::chrono::steady_clock::time_point start_time) {
102 // Create I420 buffer
103 webrtc::scoped_refptr<webrtc::I420Buffer> buffer = webrtc::I420Buffer::Create(frame.cols, frame.rows);
104
105 // Convert RGB to I420
106 libyuv::RGB24ToI420(frame.data,
107 frame.step[0],
108 buffer->MutableDataY(),
109 buffer->StrideY(),
110 buffer->MutableDataU(),
111 buffer->StrideU(),
112 buffer->MutableDataV(),
113 buffer->StrideV(),
114 frame.cols,
115 frame.rows);
116
117 // Calculate timestamp
118 const auto now = std::chrono::steady_clock::now();
119 const auto elapsed = std::chrono::duration_cast<std::chrono::microseconds>(now - start_time);
120 const int64_t timestamp_us = elapsed.count();
121
122 return webrtc::VideoFrame::Builder().set_video_frame_buffer(buffer).set_timestamp_us(timestamp_us).build();
123 }

125 static cv::Mat VideoFrame2CvMat(const webrtc::VideoFrame& frame) {
126 const auto buffer = frame.video_frame_buffer();
127 const auto i420_buffer = buffer->ToI420();
128
129 // Get frame dimensions
130 const int width = i420_buffer->width();
131 const int height = i420_buffer->height();
132
133 // Create temporary RGB buffer
134 const std::unique_ptr<uint8_t[]> rgb_data(new uint8_t[width * height * 3]);
135
136 // Convert I420 to RGB24
137 libyuv::I420ToRGB24(i420_buffer->DataY(),
138 i420_buffer->StrideY(),
139 i420_buffer->DataU(),
140 i420_buffer->StrideU(),
141 i420_buffer->DataV(),
142 i420_buffer->StrideV(),
143 rgb_data.get(),
144 width * 3, // RGB stride
145 width,
146 height);
147
148 // Create OpenCV Mat from RGB data
149 const cv::Mat rgb_mat(height, width, CV_8UC3, rgb_data.get());
150
151 // Clone the data to ensure it persists after rgb_data goes out of scope
152 cv::Mat result;
153 rgb_mat.copyTo(result);
154
155 return result;
156 }
