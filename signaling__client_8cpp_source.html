100void SignalingClient::StartInternal() {
101 ws_client_ = std::make_unique<WebSocketClient>();
102
103 ws_client_->setOnConnected([this]() {
104 MOSAIC_LOG_INFO("Connected to WebSocket server.");
105 is_connected_ = true;
106 if (latest_state_ != -1)
107 SendState(latest_state_);
108 });
109
110 ws_client_->setOnDisconnected([this]() {
111 MOSAIC_LOG_INFO("WebSocket connection disconnected.");
112 is_connected_ = false;
113 });
114
115 ws_client_->setOnError([this](const std::string& error) {
116 MOSAIC_LOG_ERROR("WebSocket connection failed: {}", error);
117 is_connected_ = false;
118 });
119
120 ws_client_->setOnMessage([this](const std::string& message) {
121 MOSAIC_LOG_INFO("WS Message Received: {}", message);
122 OnWsMessage(message);
123 });
124
125 MOSAIC_LOG_INFO("Signaling Server connecting to {}", ws_uri_);
126
127 ws_client_->connect(ws_uri_);
128 if (!ws_client_->isConnected()) {
129 MOSAIC_LOG_ERROR("Failed to connect to WebSocket server at {}", ws_uri_);
130 }
131}

133void SignalingClient::OnWsMessage(const std::string& msg) const {
134 Json::Value message;
135 const Json::CharReaderBuilder builder;
136 Json::CharReader* reader = builder.newCharReader();
137 std::string errors;
138
139 const bool parsingSuccessful = reader->parse(msg.c_str(), msg.c_str() + msg.size(), &message, &errors);
140 delete reader;
141
142 if (!parsingSuccessful) {
143 MOSAIC_LOG_ERROR("JSON parse error: {}", errors);
144 MOSAIC_LOG_ERROR("Original Message: {}", msg);
145 return;
146 }
147 OnMessage(message);
148}

150void SignalingClient::OnMessage(Json::Value const& message) const {
151 if (!message["type"]) {
152 return;
153 }
154
155 if (const std::string type = message["type"].asString(); type == "signaling.send_sdp_offer") {
156 HandleSendSdpOffer(message["data"]);
157 } else if (type == "signaling.exchange_ice_candidate") {
158 HandleExchangeIceCandidate(message["data"]);
159 } else if (type == "signaling.send_close_peer_connection") {
160 HandleCloseConnection();
161 } else if (type.rfind("authorize", 0) == 0) {
162 HandleAuthorizationMessage(type, message["data"]);
163 } else {
164 MOSAIC_LOG_DEBUG("Received unknown message type: {}", type);
165 }
166}

213webrtc::IceCandidateInterface* ResolveIceCandidate(Json::Value const& ice_candidate) {
214 MOSAIC_LOG_DEBUG("Received ICE candidate: {}", ice_candidate["candidate"].asString());
215 if (!ice_candidate["candidate"] || !ice_candidate["sdpMid"] || !ice_candidate["sdpMLineIndex"]) {
216 MOSAIC_LOG_WARNING("Received null ICE candidate string; ignoring.");
217 return nullptr;
218 }
219 std::string candidate_str = ice_candidate["candidate"].asString();
220 const std::string sdp_mid = ice_candidate["sdpMid"].asString();
221 const int sdp_mline_index = ice_candidate["sdpMLineIndex"].asInt();
222
223 webrtc::SdpParseError error;
224 webrtc::IceCandidateInterface* candidate =
225 webrtc::CreateIceCandidate(sdp_mid, sdp_mline_index, candidate_str, &error);
226 if (!candidate) {
227 MOSAIC_LOG_ERROR("Failed to create IceCandidate from string: {}, error: {}, from line {}",
228 candidate_str,
229 error.description,
230 error.line);
231 return nullptr;
232 }
233
234 return candidate;
235}
