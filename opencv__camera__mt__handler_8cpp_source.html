24void impl::OpenCVCameraMediaTrack::Start() {
25 if (IsRunning()) {
26 return;
27 }
28 if (!capture_.open(camera_id_, cv::CAP_V4L2)) {
29 return;
30 }
31 capture_.set(cv::CAP_PROP_FOURCC, cv::VideoWriter::fourcc('M', 'J', 'P', 'G'));
32
33 capture_.set(cv::CAP_PROP_FRAME_WIDTH, width_);
34 capture_.set(cv::CAP_PROP_FRAME_HEIGHT, height_);
35 capture_.set(cv::CAP_PROP_FPS, frame_rate_);
36
37 SetRunning(true);
38 SetStopFlag(false);
39 start_time_ = std::chrono::steady_clock::now();
40 frame_loop_thread_ = std::make_shared<std::thread>(std::thread([this]() {
41 while (!GetStopFlag()) {
42 try {
43 this->FrameLoop();
44 } catch (const cv::Exception& e) {
45 MOSAIC_LOG_ERROR("OpenCV error: {}", e.what());
46 }
47 std::this_thread::sleep_for(std::chrono::milliseconds(static_cast<int>(1000.0f / this->frame_rate_)));
48 }
49 }));
50}
