19std::string ExtractFileExtension(const std::string& file_path) {
20 const std::filesystem::path path(file_path);
21 std::string extension = path.extension().string();
22
23 if (extension.empty()) {
24 return "";
25 }
26
27 // Remove the leading dot from extension (e.g., ".yaml" -> "yaml")
28 if (extension[0] == '.') {
29 extension = extension.substr(1);
30 }
31
32 return extension;
33}
