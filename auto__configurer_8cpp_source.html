39void AutoConfigurer::ResolveConnectors() {
40 auto& resolver = ConnectorResolver::GetInstance();
41 for (const auto& connector_config : mosaic_config_->connectors) {
42 if (!resolver.IsSupportedType(connector_config.type)) {
43 MOSAIC_LOG_ERROR("Unsupported connector type: {}", connector_config.type);
44 continue;
45 }
46
47 const auto configurable_connector = resolver.GetConfigurableConnector(connector_config.type);
48 if (!configurable_connector) {
49 MOSAIC_LOG_ERROR("Failed to create connector: {}", connector_config.type);
50 continue;
51 }
52
53 auto config_ptr = std::make_shared<ConnectorConfig>(connector_config);
54 configurable_connector->SetConfig(config_ptr);
55
56 try {
57 configurable_connector->ValidateConfig();
58 } catch (const std::exception& e) {
59 MOSAIC_LOG_ERROR("Error validating connector config for type {}: {}", connector_config.type, e.what());
60 continue;
61 }
62 configurable_connectors_.push_back(configurable_connector);
63 }
64}

66void AutoConfigurer::ConfigureConnectors() {
67 int connector_index = 0;
68 for (const auto& configurable_connector : configurable_connectors_) {
69 if (!configurable_connector) {
70 MOSAIC_LOG_ERROR("ConfigureConnectors: Connector #{} is nullptr!", connector_index);
71 connector_index++;
72 continue;
73 }
74 configurable_connector->Configure();
75 if (const auto dc_ptr = dynamic_cast<ADCHandlerConfigurer*>(configurable_connector.get())) {
76 const auto handler = dc_ptr->GetHandler();
77 if (!handler) {
78 MOSAIC_LOG_ERROR("ConfigureConnectors: Handler is nullptr for connector #{}!", connector_index);
79 } else {
80 mosaic_connector_->AddDataChannelHandler(handler);
81 dc_handler_map_[handler->GetLabel()] = handler;
82 }
83 } else if (const auto dc_p_ptr = dynamic_cast<AParallelDCHandlerConfigurer*>(configurable_connector.get())) {
84 const auto handlers = dc_p_ptr->GetHandlers();
85 for (const auto& handler : handlers) {
86 mosaic_connector_->AddDataChannelHandler(handler);
87 dc_handler_map_[handler->GetLabel()] = handler;
88 }
89 } else if (const auto mt_ptr = dynamic_cast<AMTHandlerConfigurer*>(configurable_connector.get())) {
90 const auto handler = mt_ptr->GetHandler();
91 mosaic_connector_->AddMediaTrackHandler(handler);
92 mt_handler_map_[handler->GetTrackName()] = handler;
93 } else {
94 MOSAIC_LOG_INFO("ConfigureConnectors: Connector #{} type not recognized for handler extraction",
95 connector_index);
96 }
97
98 connector_index++;
99 }
