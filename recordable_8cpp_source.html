80 static bool IsValidRecordFilePath(const std::string& record_file_path) {
81 // 빈 문자열이면 에러
82 if (record_file_path.empty()) {
83 MOSAIC_LOG_ERROR("Record file path is empty");
84 return false;
85 }
86
87 // 디렉토리가 존재하지 않으면 생성 시도
88 if (!std::filesystem::exists(record_file_path)) {
89 try {
90 std::filesystem::create_directories(record_file_path);
91 MOSAIC_LOG_INFO("Created directory: " + record_file_path);
92 } catch (const std::filesystem::filesystem_error& ex) {
93 MOSAIC_LOG_ERROR("Failed to create directory: " + record_file_path + ", Error: " + ex.what());
94 return false;
95 }
96 }
97
98 if (!std::filesystem::is_directory(record_file_path)) {
99 MOSAIC_LOG_ERROR("Path is not a directory: " + record_file_path);
100 return false;
101 }
102
103 auto perms = std::filesystem::status(record_file_path).permissions();
104 if ((perms & std::filesystem::perms::owner_write) == std::filesystem::perms::none &&
105 (perms & std::filesystem::perms::group_write) == std::filesystem::perms::none &&
106 (perms & std::filesystem::perms::others_write) == std::filesystem::perms::none) {
107 MOSAIC_LOG_ERROR("No write permission for directory: " + record_file_path);
108 return false;
109 }
110 return true;
111 }
