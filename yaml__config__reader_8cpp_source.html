19 MosaicConfig ParseConnectorConfigs(const YAML::Node& node) {
20 MosaicConfig configs;

123 std::vector<IceServerConfig> ParseIceServers(const YAML::Node& node) {
124 std::vector<IceServerConfig> ice_servers;
125
126 for (const auto& ice_server_node : node) {
127 IceServerConfig ice_server;
128
129 if (ice_server_node["urls"]) {
130 for (const auto& url : ice_server_node["urls"]) {
131 ice_server.urls.push_back(url.as<std::string>());
132 }
133 } else {
134 throw std::invalid_argument("Ice server urls are not found.");
135 }
136
137 if (ice_server_node["username"]) {
138 ice_server.username = ice_server_node["username"].as<std::string>();
139 } else {
140 throw std::invalid_argument("Ice server username is not found.");
141 }
142
143 if (ice_server_node["credential"]) {
144 ice_server.credential = ice_server_node["credential"].as<std::string>();
145 } else {
146 throw std::invalid_argument("Ice server credential is not found.");
147 }
148
149 ice_servers.push_back(ice_server);
150 }
151
