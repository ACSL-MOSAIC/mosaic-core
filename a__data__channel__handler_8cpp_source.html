62 void RegisterDataChannelObserver(IDataChannelHandler* dc_handler) {
63 if (!dc_interface_) {
64 MOSAIC_LOG_ERROR("DataChannelInterface cannot be set for label: {}", label_);
65 throw std::runtime_error("DataChannelInterface is not set");
66 }
67
68 std::function<void(const webrtc::DataBuffer&)> onMessageLambda;
69
70 if (auto receivable_ptr = dynamic_cast<IDataChannelReceivable*>(dc_handler)) {
71 MOSAIC_LOG_DEBUG("Receivable DataChannel Detected!");
72 onMessageLambda = [receivable_ptr](const webrtc::DataBuffer& buffer) { receivable_ptr->OnMessage(buffer); };
73 } else {
74 onMessageLambda = [](const webrtc::DataBuffer&) {};
75 }
76
77 observer_ = std::make_shared<DataChannelObserver>(onMessageLambda);
78 dc_interface_->RegisterObserver(observer_.get());
79 }

81 ChannelState GetState() const {
82 if (!dc_interface_) {
83 return kUnknown;
84 }
85
86 switch (dc_interface_->state()) {
87 case webrtc::DataChannelInterface::kConnecting:
88 return kConnecting;
89 case webrtc::DataChannelInterface::kOpen:
90 return kOpen;
91 case webrtc::DataChannelInterface::kClosing:
92 return kClosing;
93 case webrtc::DataChannelInterface::kClosed:
94 return kClosed;
95 default:
96 return kUnknown;
97 }
98 }

121 void SendAsync(const webrtc::DataBuffer& buffer) const {
122 std::lock_guard<std::mutex> lock(queue_mutex_);
123 send_queue_.push(buffer);
124 ProcessQueue();
125 }
126
127 void ProcessQueue() const {
128 if (is_sending_ || send_queue_.empty()) {
129 return;
130 }
131
132 is_sending_ = true;
133 auto buffer = send_queue_.front();
134 send_queue_.pop();
135
136 MOSAIC_LOG_DEBUG("Buffer Size: {}", buffer.size());
137
138 dc_interface_->SendAsync(buffer, [this](webrtc::RTCError error) {
139 std::lock_guard<std::mutex> lock(queue_mutex_);
140 is_sending_ = false;
141
142 if (!error.ok()) {
143 MOSAIC_LOG_ERROR("SendAsync failed for label: {}, error: {}", label_, error.message());
144 }
145
146 ProcessQueue();
147 });
148 }
