72void MosaicConnector::OnDataChannel(const webrtc::scoped_refptr<webrtc::DataChannelInterface>& data_channel) const {
73 std::string label = data_channel->label();
74 MOSAIC_LOG_INFO("Data channel created with label: {}", label);
75
76 const auto it = data_channel_handlers_dict_.find(label);
77 if (it == data_channel_handlers_dict_.end()) {
78 return;
79 }
80 const std::shared_ptr<handlers::IDataChannelHandler> data_channel_handler = it->second;
81
82 if (data_channel_handler->GetState() == handlers::IDataChannelHandler::kOpen) {
83 MOSAIC_LOG_INFO("Data channel with label '{}' is already open.", label);
84 return;
85 }
86
87 data_channel_handler->SetDataChannelInterface(data_channel);
88 data_channel_handler->RegisterDataChannelObserver();
89 data_channel_handler->AfterCreate();
90}

92void MosaicConnector::CreateAllMediaTracks() const {
93 for (const auto& [track_name, media_track_handler] : media_track_handlers_dict_) {
94 auto video_source = media_track_handler->CreateVideoTrackSource();
95 if (!video_source) {
96 MOSAIC_LOG_ERROR("Failed to create video track source for {}", track_name);
97 continue;
98 }
99
100 auto video_track = peer_connection_manager_->CreateVideoTrack(video_source, track_name);
101 if (!video_track) {
102 MOSAIC_LOG_ERROR("Failed to create video track for {}", track_name);
103 continue;
104 }
105
106 auto add_track_result = peer_connection_manager_->AddTrack(video_track, {track_name});
107 if (!add_track_result) {
108 MOSAIC_LOG_ERROR("Failed to add video track for {}", track_name);
109 }
110 }
111}
