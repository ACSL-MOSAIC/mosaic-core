45void WebSocketClient::connectInternal() {
46 // WebSocket 클라이언트 설정
47 websocket_client_config config;
48
49 // WSS 연결을 위한 SSL 설정
50 if (m_uri.find("wss://") == 0) {
51 config.set_validate_certificates(false); // 개발용 - 실제 환경에서는 true로 설정
52 }
53
54 // 새로운 클라이언트 생성
55 m_client = std::make_shared<websocket_callback_client>(config);
56
57 // 이벤트 핸들러 설정
58 setupEventHandlers();
59
60 try {
61 // 연결 시도
62 const auto res = m_client->connect(m_uri)
63 .then([this] {
64 m_connected = true;
65 MOSAIC_LOG_INFO("WebSocket connected to {}", m_uri);
66
67 if (m_onConnected) {
68 m_onConnected();
69 }
70 })
71 .wait();
72 if (res != pplx::task_status::completed) {
73 throw std::runtime_error("Failed to connect to WebSocket server");
74 }
75 } catch (const std::exception& e) {
76 MOSAIC_LOG_ERROR("Connection failed: {}", e.what());
77 }
78}

137void WebSocketClient::handleMessage(const websocket_incoming_message& message) const {
138 try {
139 if (message.message_type() == websocket_message_type::ping) {
140 MOSAIC_LOG_DEBUG("Received ping message");
141 auto pong_message = websocket_outgoing_message();
142 pong_message.set_pong_message("pong");
143 m_client->send(pong_message).wait();
144 return; // Ping 메시지는 Pong으로 응답하고 무시
145 }
146 if (message.message_type() == websocket_message_type::pong) {
147 MOSAIC_LOG_DEBUG("Received pong message");
148 return; // Pong 메시지는 무시
149 }
150 if (message.message_type() == websocket_message_type::close) {
151 m_client->close().wait();
152 MOSAIC_LOG_DEBUG("Received close message from server");
153 return;
154 }
155 if (message.message_type() == websocket_message_type::text_message) {
156 auto ts = message.extract_string()
157 .then([this](const std::string& body) {
158 if (m_onMessage) {
159 m_onMessage(body);
160 }
161 })
162 .wait();
163 if (ts != pplx::task_status::completed) {
164 m_onError("Task did not complete successfully while handling message.");
165 }
166 }
167 } catch (const std::exception& e) {
168 const std::string error = "Message handling error: " + std::string(e.what());
169 if (m_onError) {
170 m_onError(error);
171 }
172 }
173}

175void WebSocketClient::handleClose(websocket_close_status close_status,
176 const std::string& reason,
177 const std::error_code&) {
178 m_connected = false;
179
180 /*
181 *enum class websocket_close_status
182 {
183 normal = 1000,
184 going_away = 1001,
185 protocol_error = 1002,
186 unsupported = 1003, // or data_mismatch
187 abnormal_close = 1006,
188 inconsistent_datatype = 1007,
189 policy_violation = 1008,
190 too_large = 1009,
191 negotiate_error = 1010,
192 server_terminate = 1011,
193 };
194 */
195 MOSAIC_LOG_INFO("WebSocket closed. Status: {} Reason: {} Error Code: {}",
196 static_cast<int>(close_status),
197 reason,
198 std::error_code().message());
199
200 if (m_onDisconnected) {
201 m_onDisconnected();
202 }
203
204 // Reconnection logic start
205 std::thread reconnect_thread([this] {
206 MOSAIC_LOG_ERROR("Reconnecting to WebSocket server in 5 seconds...");
207
208 std::this_thread::sleep_for(std::chrono::seconds(5)); // Wait before reconnecting
209 this->connectInternal();
210 });
211 reconnect_thread.detach();
212}
